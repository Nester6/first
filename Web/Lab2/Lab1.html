Програмування

amount  = int(input('Введіть суму:'))
period  = int(input('На скільки років:'))
percent  = int(input('Введіть відсоток:'))

if amount == 0 or period == 0 or percent == 0:
  print('Error')
else:
  balance = amount
  for i in range(1, period +1):
    amount  *= (1 + percent / 100)
    print(f'{i} рік: {amount:.2f} (після {percent}%)')
  profit = amount - balance
  print(f"Чистий прибуток за {period} років: {profit:.2f}")
